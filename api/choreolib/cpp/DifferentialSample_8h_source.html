115 bool operator==(const DifferentialSample& other) const {
116 constexpr double epsilon = 1e-6;
117
118 auto compare_units = [epsilon](const auto& a, const auto& b) {
119 using UnitType =
120 std::remove_const_t<std::remove_reference_t<decltype(a)>>;
121 return units::math::abs(a - b) < UnitType(epsilon);
122 };
123
124 return compare_units(timestamp, other.timestamp) &&
125 compare_units(x, other.x) && compare_units(y, other.y) &&
126 compare_units(heading, other.heading) &&
127 compare_units(vl, other.vl) && compare_units(vr, other.vr) &&
128 compare_units(al, other.al) && compare_units(ar, other.ar) &&
129 compare_units(fl, other.fl) && compare_units(fr, other.fr);
