26 AutoBindings Bind(std::string_view name, frc2::CommandPtr cmd) && {

31 void Merge(AutoBindings&& other) {
32 for (auto& [key, value] : other.bindings) {
33 bindings.emplace(std::move(key), std::move(value));

35 other.bindings.clear();

46template <choreo::TrajectorySample SampleType>
47class AutoFactory {
48 public:
49 AutoFactory(std::function<frc::Pose2d()> poseSupplier,
50 ChoreoControllerFunction<SampleType> controller,
51 std::function<void(frc::ChassisSpeeds)> outputChassisSpeeds,
52 std::function<bool()> mirrorTrajectory,
53 const frc2::Subsystem& driveSubsystem, AutoBindings bindings,
54 std::optional<TrajectoryLogger> trajectoryLogger)
55 : poseSupplier{std::move(poseSupplier)},
56 controller{controller},
57 outputChassisSpeeds{std::move(outputChassisSpeeds)},

59 driveSubsystem{driveSubsystem},
60 autoBindings{std::move(bindings)},
61 trajectoryLogger{std::move(trajectoryLogger)} {}

63 AutoLoop<SampleType> NewLoop() const {
64 // Clear cache in simulation to allow a form of "hot-reloading" trajectories
65 if (RobotBase::IsSimulation()) {
66 ClearCache();
67 }
68
69 return AutoLoop<SampleType>();
70 }
71
72 AutoTrajectory<SampleType> Trajectory(std::string_view trajectoryName,
73 AutoLoop<SampleType> loop) const {
74 std::optional<choreo::Trajectory<SampleType>> optTraj =
75 Choreo::LoadTrajectory<SampleType>(trajectoryName);
76 choreo::Trajectory<SampleType> trajectory;
77 if (optTraj.has_value()) {
78 trajectory = optTraj.value();
79 } else {
80 FRC_ReportError(frc::warn::Warning, "Could not load trajectory: {}",
81 trajectoryName);
82 }
83 AutoTrajectory<SampleType> autoTraj{
84 trajectoryName, trajectory, poseSupplier,
85 controller, outputChassisSpeeds, mirrorTrajectory,
86 trajectoryLogger, driveSubsystem, loop.GetLoop(),
87 autoBindings, loop.OnNewTrajectory};
88 loop.AddTrajectory(autoTraj);
89 return autoTraj;
90 }

92 AutoTrajectory<SampleType> Trajectory(std::string_view trajectoryName,
93 int splitIndex,
94 AutoLoop<SampleType> loop) const {
95 std::optional<choreo::Trajectory<SampleType>> optTraj =
96 Choreo::LoadTrajectory<SampleType>(trajectoryName);
97 choreo::Trajectory<SampleType> trajectory;
98 if (optTraj.has_value()) {
99 trajectory = optTraj.value();

101 FRC_ReportError(frc::warn::Warning, "Could not load trajectory: {}",
102 trajectoryName);
103 }
104 return Trajectory(trajectory, loop);

107 AutoTrajectory<SampleType> Trajectory(

119 void Bind(std::string_view name, frc2::CommandPtr cmd) {
