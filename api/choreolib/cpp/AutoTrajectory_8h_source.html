58 frc2::CommandPtr Cmd() const {
59 if (trajectory.samples.size() == 0) {
60 return frc2::cmd::RunOnce([] {
61 FRC_ReportError(frc::warn::Warning,
62 fmt::format("Trajectory {} has no samples", name);
63 })
64 .WithName("Trajectory_" + name);
65 }
66 return frc2::FunctionalCommand(
67 [this] { return CmdInitiazlize(); },
68 [this] { return CmdExecute(); }, [this] { return CmdEnd(); },
69 [this] { return CmdIsFinished(); }, {driveSubsystem})
70 .WithName("Trajectory_" + name);
71 }

99 frc2::Trigger AtTime(units::second_t timeSinceStart) {
100 if (timeSinceStart < 0_s) {
101 FRC_ReportError(frc::warn::Warning,
102 "Trigger time cannot be negative for" + name);
103 return offTrigger;
104 }
105
106 if (timeSinceStart > TotalTime()) {
107 FRC_ReportError(
108 frc::warn::Warning,
109 "Trigger time cannout be greater than total trajectory time for " +
110 name);
111 return offTrigger;
112 }
113
114 return frc2::Trigger(
115 loop, [this, timeSinceStart, lastTimestamp = timer.Get()]() mutable {
116 units::second_t nowTimestamp = timer.Get();
117 bool shouldTrigger =
118 lastTimestamp < nowTimestamp && nowTimestamp >= timeSinceStart;
119 lastTimestamp = nowTimestamp;
120 return shouldTrigger;
121 });
122 }

124 frc2::Trigger AtTime(std::string_view eventName) {
125 bool foundEvent = false;
126 frc2::Trigger trig = offTrigger;
127
128 for (const auto& event : trajectory.GetEvents(eventName)) {
129 trig = frc2::Trigger(trig || AtTime(event.timestamp));
130 foundEvent = true;
131 }
132
133 if (!foundEvent) {
134 FRC_ReportError(frc::warn::Warning, "Event \"{}\" not found for {}",
135 eventName, name);
136 }
137
138 return trig;
139 }
140
141 frc2::Trigger AtPose(std::string_view eventName,
142 units::meter_t tolerance = DEFAULT_TOLERANCE) {
143 bool foundEvent = false;
144 frc2::Trigger trig = offTrigger;
145
146 for (const auto& event : trajectory.GetEvents(eventName)) {
147 frc::Pose2d pose =
148 trajectory.SampleAt<Year>(event.timestamp, mirrorTrajectory())
149 .GetPose();
150 trig = frc2::Trigger(trig || AtPose(pose, tolerance));
151 foundEvent = true;
152 }
153
154 if (!foundEvent) {
155 FRC_ReportError(frc::warn::Warning, "Event \"{}\" not found for {}",
156 eventName, name);
157 }
158
159 return trig;
160 }

162 frc2::Trigger AtTimeAndPlace(std::string_view eventName,
163 units::meter_t tolerance = DEFAULT_TOLERANCE) {
164 return frc2::Trigger{AtTime(eventName) && AtPose(eventName, tolerance)};
165 }

167 private:
168 void OnNewTrajectory() {
169 isDone = false;
170 isActive = false;
171 }

173 units::second_t TimeIntoTraj() { return timer.Get() + timeOffset; }
174
175 units::second_t TotalTime() { return trajectory.GetTotalTime(); }
176
177 void LogTrajectory(bool starting) {
178 if (trajectoryLogger.has_value()) {
179 trajectoryLogger.value()(trajectory.GetPoses(), starting);
180 }
181 }
182
183 void CmdInitiazlize() {
184 newTrajectoryCallback();
185 timer.Restart();
186 isDone = false;
187 isActive = true;
188 timeOffset = 0.0;
189 LogTrajectory(true);
190 }
191
192 void CmdExecute() {
193 SampleType sample =
194 trajectory.SampleAt<Year>(TimeIntoTraj(), mirrorTrajectory());
195 frc::ChassisSpeeds chassisSpeeds = DEFAULT_CHASSIS_SPEEDS;
196 chassisSpeeds = controller(poseSupplier(), sample);
197 outputChassisSpeeds(chassisSpeeds);
198 }
199
200 void CmdEnd(bool interrupted) {
201 timer.Stop();
202 if (interrupted) {
203 outputChassisSpeeds(frc::ChassisSpeeds{});
204 } else {
205 outputChassisSpeeds(trajectory.GetFinalSample().GetChassisSpeeds());
206 }
207 isDone = true;
208 isActive = false;
209 LogTrajectory(false);
210 }
211
212 bool CmdIsFinished() { return TimeIntoTraj() > TotalTime(); }

214 frc2::Trigger AtPose(frc::Pose2d pose, units::meter_t tolerance) {
215 frc::Translation2d checkedTrans =
216 mirrorTrajectory()
217 ? frc::Translation2d{16.5410515_m - pose.Translation().X(),
218 pose.Translation().Y}
219 : pose.Translation();
220 return frc2::Trigger{
221 loop, [this, checkedTrans, tolerance] {
222 frc::Translation2d currentTrans = poseSupplier().Translation();
223 return currentTrans.Distance(checkedTrans) < tolerance;
224 }};
225 }

239 bool isDone = false;
240 bool isActive = false;

242 frc2::Trigger offTrigger;
243};
