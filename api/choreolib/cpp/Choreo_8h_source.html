45 static ProjectFile GetProjectFile() {
46 if (LAZY_PROJECT_FILE.has_value()) {
47 return LAZY_PROJECT_FILE.value();
48 }
49
50 std::vector<std::filesystem::path> matchingFiles;
51
52 try {
53 for (const auto& entry :
54 std::filesystem::directory_iterator(CHOREO_DIR)) {
55 if (std::filesystem::is_regular_file(entry) &&
56 entry.path().extension() == TRAJECTORY_FILE_EXTENSION) {
57 matchingFiles.push_back(entry.path());
58 }
59 }
60
61 if (matchingFiles.size() == 0) {
62 FRC_ReportError(frc::warn::Warning,
63 "Could not find file in deploy directory!");
64 } else if (matchingFiles.size() > 1) {
65 FRC_ReportError(frc::warn::Warning,
66 "Found multiple project files in deploy directory!");
67 }
68
69 auto fileBuffer = wpi::MemoryBuffer::GetFile(matchingFiles[0].string());
70 if (!fileBuffer) {
71 FRC_ReportError(frc::warn::Warning,
72 "Could not open choreo project file");
73 }
74
75 wpi::json json = wpi::json::parse(fileBuffer.value()->GetCharBuffer());
76 std::string version = json["version"];
77 if (kSpecVersion != version) {
78 throw fmt::format(".chor project file: Wrong version {}. Expected {}",
79 version, kSpecVersion);
80 }
81 ProjectFile resultProjectFile;
82 from_json(json, resultProjectFile);
83 LAZY_PROJECT_FILE = resultProjectFile;
84 } catch (const std::filesystem::filesystem_error&) {
85 FRC_ReportError(frc::warn::Warning, "Error finding choreo directory!");
86 } catch (const wpi::json::exception&) {
87 FRC_ReportError(frc::warn::Warning, "Error parsing choreo project file!");
88 }
89 return LAZY_PROJECT_FILE.value();
90 }

102 template <TrajectorySample SampleType>
103 static std::optional<Trajectory<SampleType>> LoadTrajectory(
104 std::string_view trajectoryName) {
105 if (trajectoryName.ends_with(TRAJECTORY_FILE_EXTENSION)) {
106 trajectoryName = trajectoryName.substr(
107 0, trajectoryName.size() - TRAJECTORY_FILE_EXTENSION.size());
108 }
109
110 std::string trajectoryFileName = fmt::format(
111 "{}/{}{}", CHOREO_DIR, trajectoryName, TRAJECTORY_FILE_EXTENSION);
112
113 auto fileBuffer = wpi::MemoryBuffer::GetFile(trajectoryFileName);
114 if (!fileBuffer) {
115 FRC_ReportError(frc::warn::Warning, "Could not find trajectory file: {}",
116 trajectoryName);
117 return {};
118 }
119
120 try {
121 return LoadTrajectoryString<SampleType>(
122 std::string{fileBuffer.value()->GetCharBuffer().data(),
123 fileBuffer.value()->size()},
124 trajectoryName);
125 } catch (wpi::json::parse_error& ex) {
126 FRC_ReportError(frc::warn::Warning, "Could not parse trajectory file: {}",
127 trajectoryName);
128 FRC_ReportError(frc::warn::Warning, "{}", ex.what());
129 return {};
130 }
131 return {};
132 }

162 template <TrajectorySample SampleType>
163 class TrajectoryCache {
164 public:
178 static std::optional<Trajectory<SampleType>> LoadTrajectory(
179 std::string_view trajectoryName) {
180 if (cache.contains(std::string{trajectoryName})) {
181 return cache[std::string{trajectoryName}];
182 } else {
183 cache[std::string{trajectoryName}] =
184 Choreo::LoadTrajectory<SampleType>(trajectoryName);
185 return cache[std::string{trajectoryName}];
186 }
187 }
188
204 static std::optional<Trajectory<SampleType>> LoadTrajectory(
205 std::string_view trajectoryName, int splitIndex) {
206 std::string key = fmt::format("{}.:.{}", trajectoryName, splitIndex);
207
208 if (!cache.contains(key)) {
209 if (cache.contains(std::string{trajectoryName})) {
210 cache[key] =
211 cache[std::string{trajectoryName}].value().GetSplit(splitIndex);
212 } else {
213 auto possibleTrajectory = LoadTrajectory(trajectoryName);
214 cache[std::string{trajectoryName}] = possibleTrajectory;
215
216 if (possibleTrajectory.has_value()) {
217 cache[key] = possibleTrajectory.value().GetSplit(splitIndex);
218 }
219 }
220 }
221
222 return cache[key];
223 }
224
228 static void Clear() { cache.clear(); }
229
230 private:
231 static inline std::unordered_map<std::string,
232 std::optional<Trajectory<SampleType>>>
233 cache;
234 };
